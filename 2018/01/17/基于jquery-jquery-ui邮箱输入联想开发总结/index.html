基于jquery/jquery-ui邮箱输入联想开发总结

发表于 2018-01-17

最近用到了jquery ui的autocomplete 自动补全，感觉jq ui真的很强大,同时自己手写了基于jq的邮箱联想,跟大家分享一下，希望对初学者有用。目前仍有不完善地方(移动端不友好,手写jq滚动条没做处理),我也在努力改进中,希望大家多批评指正,废话不多说,直接上代码:

一.基于jQuery UI autocomplete

HTML:

<div class="inputs">
    <span class="mails">邮箱:</span>
    <input class="required"
           id="username"
           size="25"
           tabindex="1"
           type="text"
           placeholder="请输入用户名"
           accesskey="n"
           onfocus="placeholder = ''"
           autocomplete="off" name="username" value=""/>
</div>

引入jquery/jquery-ui:

<script src="./public/jquery.js"></script>
<script src="./public/jquery-ui.min.js"></script>

js逻辑代码:

<script>
    var mail_List = [
        'yahoo.com',
        'msn.com',
        'hotmail.com',
        'aol.com',
        'ask.com',
        'live.com',
        'qq.com',
        '0355.net',
        '163.com',
        '163.net',
        '263.net',
        '3721.net',
        'yeah.net',
        'googlemail.com',
        'mail.com'
    ];
    var usernametip = "\u8BF7\u8F93\u5165\u7528\u6237\u540D";
    $("#username").blur(function(){
        $("#username").attr("placeholder", usernametip);
    });

    // depend on jquery/jquery-ui.js(necessary)
    $("#username").autocomplete({ 
        width:200,
        delay: 10, 
        autoFocus: true, //设置为true 时，第一个项目会自动被选定。
        open: function() {
          $('.ui-autocomplete').css('width','200px');
        },
        source: function (request, response) {  
            var hosts = mail_List;//邮箱域名集合  
            var term = request.term; //获取用户输入的内容；  
            var name = term;  //邮箱的用户名  
            var host = "";   //邮箱的域名 例如qq.com  
            var ix = term.indexOf('@'); //@的位置  
            var result = []; //最终呈现的邮箱列表  

            //当用户输入的数据（email）里存在@的时候，就重新给用户名和域名赋值  
            if (ix > -1) {  
                name = term.slice(0, ix);  
                host = term.slice(ix + 1);  
            }  
            if (name) { 
                var getHosts = []; 
                getHosts=  host ? ($.grep(hosts, function (val) { return val.indexOf(host) > -1 })) : hosts;   
                result = $.map(getHosts, function (val) { 
                    return name + "@" + val;  
                });                  
            }            
            //result.unshift(term); 
            //unshift方法的作用是：将一个或多个新元素添加到数组开始，数组中的元素自动后移，返回数组新长度  

            response(result);  
        },
        response: function (event, ui) {
            // var uls = ui.content;
            // $.each(uls,function(index,val) {
            //   if (!val.flag){
            //     val.value = val.label.substring(0,val.label.lastIndexOf("-")-1);
            //     val.flag = true;
            //   }
            // })
        }
    });
</script>

二.基于jQuery

HTML:

<div class="inputs">
    <span class="mails">邮箱:</span>
    <input class="required"
           id="username"
           size="25"
           tabindex="1"
           type="text"
           placeholder="请输入用户名"
           accesskey="n"
           onfocus="placeholder = ''"
           autocomplete="off" name="username" value=""/>
    <div id="mailBox" class="hide">
        <ul class="mailList"></ul>
    </div>
</div>

引入jquery

<script src="./public/jquery.js"></script>

js代码:

<script>
    var mail_List = [
        '@yahoo.com',
        '@msn.com',
        '@hotmail.com',
        '@aol.com',
        '@ask.com',
        '@live.com',
        '@qq.com',
        '@0355.net',
        '@163.com',
        '@163.net',
        '@263.net',
        '@3721.net',
        '@yeah.net',
        '@googlemail.com',
        '@mail.com'
    ];
    var usernametip = "\u8BF7\u8F93\u5165\u7528\u6237\u540D";
    $("#username").blur(function(){
        $("#username").attr("placeholder", usernametip);
        $('#mailBox').addClass('hide');
    });

    var lis = $.map(mail_List,function(val) {
      return '<li>'+val+'</li>';
    }).join('');
    $('.mailList').html(lis);

    //onclick li
    $('.mailList').delegate('li','click',function() {
      var thisTxt = $(this).text();
      //var textStr = thisTxt.substring(0,thisTxt.lastIndexOf("-")-1);
      $('#username').val(thisTxt);
      $('#username').blur();
      $('#mailBox').addClass('hide');
    });
    //preventDefault input focus
    $('.mailList').delegate('li','mousedown', false);

    $('.mailList').delegate('li','mouseenter',function(){
      $(this).addClass('active').siblings().removeClass('active');
    })

    // $('.mailList').delegate('li','mouseleave',function(){
    //    $(this).removeClass('active');
    // })

    $('#username').keyup(function(e) {
      if (e.keyCode != 40 && e.keyCode != 38 && e.keyCode != 13) {
        $('.mailList').html(lis);
      }

      var txt = $(this).val();
      var suffix = "";
      var val;
      var contents;
      if (txt != '' && txt != ' ') {
        $('#mailBox').removeClass('hide');
      }else {
        $('#mailBox').addClass('hide');
      }
      if(txt.indexOf("@") >= 0) {
        suffix = txt.substring(txt.indexOf("@")-1);
        txt = txt.substring(0,txt.indexOf("@"));
      }
      //sync li
      $('.mailList li').each(function() {
        val = $(this).text();
        val = val.substring(val.indexOf("@"));
        contents = txt + val;
        $(this).text(contents);
      })
      //filter @ after content
      if (suffix != "") {
        var reg = new RegExp(suffix,"i");
        $('.mailList li').each(function() {
          if(reg.test($(this).text())) {
            //$(this).css("display",'block');
          }else {
            $(this).remove();
          }
        })
        //if not filtered to hide
        if($('.mailList').height() == 0) {
          $('#mailBox').addClass('hide');
        }
      }
      if (e.keyCode != 40 && e.keyCode != 38 && e.keyCode != 13) {
         $('.mailList li').eq(0).addClass('active').nextAll().removeClass('active');
      }

      if (e.keyCode == 38) {
        selectSearch('prev')
        // 向下键
      } else if (e.keyCode == 40) {
        selectSearch('next')
        // 回车键
      } else if (e.keyCode == 13) {
        var thisTxt = $('.active').text();
        //var textStr = thisTxt.substring(0,thisTxt.lastIndexOf("-")-1);
        $('#username').val(thisTxt);
        $('#username').blur();
        $('#mailBox').addClass('hide');

      }
    })

    function selectSearch(type){
      var index;
      if (type == 'next'){
        if ($('.mailList li').hasClass('active')) {
           $('.active').next().addClass('active').siblings().removeClass('active');
         }else {
          $('.mailList li').eq(0).addClass('active').siblings().removeClass('active');
         }
      }else{
        $('.active').prev().addClass('active').siblings().removeClass('active');
      }
    }
</script>
